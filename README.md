```
	./testify_replace.sh replace_dir(file)  pattern_file
	./testify_replace.sh replace_dir(file)
```

	./testify_replace.sh client_test.go testify.pattern
	./testify_replace.sh client_test.go

	./testify_replace.sh ./cdc/kv
